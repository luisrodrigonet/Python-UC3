### 3.2 Arquivo: `conta.py`

Vamos separar a classe `ContaBancaria` no arquivo `conta.py`.

```python
class ContaBancaria:
    def __init__(self, titular, saldo_inicial=0):
        self.titular = titular
        self.saldo = saldo_inicial
        self.historico = []

    def depositar(self, valor):
        if valor > 0:
            self.saldo += valor
            self.historico.append(f"Dep√≥sito: +R$ {valor:.2f}")
            print(f"Dep√≥sito de R$ {valor:.2f} realizado com sucesso.")
        else:
            print("Valor de dep√≥sito inv√°lido.")

    def sacar(self, valor):
        if valor > 0 and valor <= self.saldo:
            self.saldo -= valor
            self.historico.append(f"Saque: -R$ {valor:.2f}")
            print(f"Saque de R$ {valor:.2f} realizado com sucesso.")
        elif valor > self.saldo:
            print("Saldo insuficiente para saque.")
        else:
            print("Valor de saque inv√°lido.")

    def transferir(self, outra_conta, valor):
        if valor > 0 and valor <= self.saldo:
            self.saldo -= valor
            outra_conta.saldo += valor
            self.historico.append(f"Transfer√™ncia para {outra_conta.titular}: -R$ {valor:.2f}")
            outra_conta.historico.append(f"Recebido de {self.titular}: +R$ {valor:.2f}")
            print(f"Transfer√™ncia de R$ {valor:.2f} para {outra_conta.titular} realizada.")
        elif valor > self.saldo:
            print("Saldo insuficiente para transfer√™ncia.")
        else:
            print("Valor de transfer√™ncia inv√°lido.")

    def mostrar_saldo(self):
        print(f"Saldo de {self.titular}: R$ {self.saldo:.2f}")

    def mostrar_historico(self):
        print(f"\nHist√≥rico de transa√ß√µes de {self.titular}:")
        for transacao in self.historico:
            print(f"- {transacao}")
        print()
```

## :four: Encapsulamento 

Para evitar que o saldo seja alterado diretamente, podemos usar **encapsulamento** com **atributos privados**:

```python

    def __init__(self, titular, saldo_inicial=0):
        self.titular = titular
        self.__saldo = saldo_inicial  # Atributo privado
        self.historico = []

    def get_saldo(self):
        return self.__saldo

    def depositar(self, valor):
        if valor > 0:
            self.__saldo += valor
            self.historico.append(f"Dep√≥sito: +R$ {valor:.2f}")
```

> Agora o saldo s√≥ pode ser alterado pelos m√©todos da classe.
